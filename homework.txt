        self.speed = speed
        self.color = color
        self.name = name
        self.is_police = bool()

    def go(self):
        return print(self.name, 'is going with', self.speed, 'km/h')

    def stop(self):
        return print(self.name, 'has stopped')

    def turn(self):
        direction = input('which direction do you want to go ?\n ')

def folder_change(x):
    os.chdir(x)

def folder_index(x):
    print(os.listdir(x))

def folder_delete(x):
    os.removedirs(x)

def folder_create(x):
    os.mkdir(x)

a = int(input('please make your choice: '
      '1. Перейти в папку, '
      '2. Просмотреть содержимое текущей папки, '
      '3. Удалить папку, '
      '4. Создать папку\n'))

x = input('please write folder name ')

if a == 1:
    folder_change(x)
    print("Успешно перешел")
elif a == 2:
    folder_index(x)
elif a == 3:
    folder_delete(x)
    print("Успешно удалено")
elif a == 4:
    folder_create(x)
    print('Успешно создано')
else:
    print("Невозможно создать/удалить/перейти")

    if not dir_name:
        print("Необходимо указать имя директории вторым параметром")
        return
    dir_path = os.path.join(os.getcwd(), dir_name)
    try:
        os.mkdir(dir_path)
        print('директория {} создана'.format(dir_name))

        print('директория {} уже существует'.format(dir_name))
